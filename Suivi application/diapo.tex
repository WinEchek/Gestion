\section{WinEchek}
\begin{frame}
	\frametitle{WinEchek}
	\begin{block}{Rappel des enjeux}
	Développement d'une applicaiton de jeu d'échec afin de permettre une montée en connaissance, notament avec le langage C\#.
	\end{block}

\end{frame}

\begin{frame}
	\frametitle{WinEchek}
	\begin{block}{Description du projet}
	Un jeu d'échec permettant de :
	\begin{itemize}
	 \item Jouer en local contre un autre joueur
	 \item Jouer en local contre une intelligence artificielle
	 \item Jouer en réseau contre un autre joueur
	\end{itemize}

	\end{block}
\end{frame}


\section{Menu}
\begin{frame}
	\frametitle{Menu}
	\begin{block}{Un menu principal}
		\begin{itemize}
			\item Jouer en local : Pour jouer une partie locale contre un autre joueur
			\item Jouer en réseau : Pour jouer une partie en réseau
			\item Jouer contre l'ordinateur : Pour jouer une partie contre un ordinateur
			\item Paramètres : Permet d'accéder aux paramètres de WinEchek
			\item Contribuer : Pour participer au développement de WinEchek
		\end{itemize}
	\end{block}
