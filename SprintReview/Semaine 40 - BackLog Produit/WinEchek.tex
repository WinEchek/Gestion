\section{Backlog produit}
\begin{itemize}
 \item Gestion de l'affichage et des contrôlse de l'utilisateur (IHM)
 \begin{itemize}
  \item{\textcolor{red}{Haute}} L'utilisateur pourra lancer une partie contre un adversaire de type humains sur la même machine
  \item{\textcolor{red}{Haute}} L'utilisateur pourra lancer une partie contre un adversaire en local
  \item{\textcolor{orange}{Moyenne}} L'utilisateur pourra lancer une partie contre un adversaire en réseau
  \item{\textcolor{green}{Basse}} L'utilisateur pourra lancer une partie contre une intelligence artificielle
  \item{\textcolor{red}{Haute}} Présence du plateau et des pièces lors de l'affichage d'une partie
  \item{\textcolor{orange}{Moyenne}} Présence d'un timer dans l'affichage d'une partie
  \item{\textcolor{orange}{Moyenne}} Présence d'un historique des coups dans l'affichage d'une partie
  \item{\textcolor{red}{Haute}} Présence d'un affichage des pièces prises lors d'une partie
  \item{\textcolor{red}{Haute}} L'utilisateur pourra contrôler le déplacement de ses pièces grâce à la souris
  \item{\textcolor{orange}{Moyenne}} Afficher les coups possibles lorsque le joueur sélectionne une pièce
  \item{\textcolor{green}{Basse}} Afficher un menu pour modifier des options d'affichage ou sauvegarder un historique de partie
 \end{itemize}
 \item Moteur de jeu et contrôle des parties
 \begin{itemize}
  \item{\textcolor{red}{Haute}} Déroulement en tour par tout
  \item{\textcolor{red}{Haute}} Contrôle des déplacements possibles
  \item{\textcolor{red}{Haute}} Contrôle des positions d'échec
  \item{\textcolor{red}{Haute}} Contrôle de prise des pièces
  \item{\textcolor{red}{Haute}} Contrôle de la promotion d'un pion en une autre pièce prise par l'adversaire
  \item{\textcolor{red}{Haute}} Contrôle de fin de partie (échec et mat)
 \end{itemize}
 \item Gestion des parties en réseau
 \begin{itemize}
  \item{\textcolor{red}{Haute}} Permettre à l'utilisateur de jouer en réseau avec quelqu'un d'autre soit en local (sur un autre ordinateur)
  \item{\textcolor{orange}{Moyenne}} Permettre à l'utilisateur de jouer en réseau avec quelqu'un d'autre à travers internet
 \end{itemize}
 \item Mise en place d'un joueur virtuel
 \begin{itemize}
  \item{\textcolor{green}{Basse}} L'utilisateur jouera contre un joueur virtuel (intelligence artificielle) qui proposera plusieurs niveaux de difficulté
 \end{itemize}




 

\end{itemize}
