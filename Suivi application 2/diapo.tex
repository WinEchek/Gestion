\section{WinEchek}
\begin{frame}
	\frametitle{WinEchek}
	\begin{block}{Rappel des enjeux}
	\begin{itemize}
	 \item Développer une application de jeu d'échec afin de permettre une montée en connaissance, notament avec le langage C\#, aborder les intelligence artificelles et la gestion du réseau.
	 \item Arriver à présenter un livrable final conforme aux attentes de début de projet.
	 \item Réaliser et suivre un projet d'ampleur moyenne sur plusieurs mois.
	\end{itemize}
	\end{block}

\end{frame}

\begin{frame}
	\frametitle{WinEchek}
	\begin{block}{Description du projet}
	Un jeu d'échec permettant de :
	\begin{itemize}
	 \item Jouer en local contre un autre joueur
	 \item Jouer en local contre une intelligence artificielle
	 \item Jouer en réseau contre un autre joueur
	\end{itemize}

	\end{block}
\end{frame}
